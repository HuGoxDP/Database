	Скласти SQL-script, який виконує:

a. Видалення попередньої версії бази даних
DROP DATABASE IF EXISTS abc;
---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
b. Створення бази даних 
CREATE DATABASE IF NOT EXISTS abc;
---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
c. Створення таблиці на основі універсального відношення. Команда для створення таблиці повинна містити головний ключ, обмеження типу null / not null, default, check.   ЗВЕРНІТЬ УВАГУ!!! Усі поля повинні мати назви латиницею!
DROP TABLE IF EXISTS homework1;
---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
CREATE TABLE IF NOT EXISTS homework1(
Id INT NOT NULL AUTO_INCREMENT PRIMARY KEY ,
Code INT NOT NULL ,
New BOOLEAN NOT NULL DEFAULT '0',
Title VARCHAR(100) NOT NULL,
Price FLOAT(30) NOT NULL DEFAULT'0',
Edition VARCHAR(100) NOT NULL,
Pages INT NOT NULL,
Format VARCHAR(40) DEFAULT 'Nothing',
Data DATE DEFAULT NULL,
Circulation INT NOT NULL CHECK(Circulation <= 5000),
Topic VARCHAR(40) NOT NULL,
Category VARCHAR(40) NOT NULL
);
---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
d. Створення додаткового індексу.
CREATE INDEX index_new ON homework1(New);  
---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
e. Завантаження даних в таблицю
LOAD DATA INFILE 'TXT.txt '
INTO TABLE homework1;
---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
	Скласти SQL-script, який виконує:

a. Додавання в таблицю нового текстового поля "Автор", розміром 15 символів
ALTER TABLE homework1 ADD Author VARCHAR(15);
---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
b. Збільшення розміру текстового поля "Автор" до 20 символів
ALTER TABLE homework1 MODIFY COLUMN Author VARCHAR(20);
---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
c. Видалення текстового поля "Автор" з таблиці
ALTER TABLE homework1 DROP COLUMN Author;

d. Зміна порядку сортування індексу і призначення його унікальності
SELECT * FROM homework1 ORDER BY ID DESC;
---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
e. Видалення індексу
DROP INDEX unique_index ON homework1;
